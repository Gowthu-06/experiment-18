Grade.java code=
public class Grade {

    // Returns a letter grade based on the average
    static char gradeFunction(double avg) {
        if (avg >= 90)
            return 'A';
        else if (avg >= 80)
            return 'B';
        else if (avg >= 70)
            return 'C';
        else if (avg >= 60)
            return 'D';
        else
            return 'F';
    }

    public static void main(String[] args) {

        // Fixed grades without user input
        double grade1 = 85;
        double grade2 = 90;
        double grade3 = 78;
        double grade4 = 88;
        double grade5 = 92;

        int count = 5;

        double sum = grade1 + grade2 + grade3 + grade4 + grade5;
        double avg = sum / count;

        System.out.println("Average: " + avg);
        System.out.println("Letter Grade: " + gradeFunction(avg));
    }
}
